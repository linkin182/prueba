select docfac.nodoc, Tipo_Pac.DSC as Seguro, docfac.paciente, docfac.codtipaten,concepto.concepto, servicio.servicio from docfac left join detfac on docfac.nodoc=detfac.nodoc AND docfac.noserie=detfac.noserie AND docfac.codtipdoc=detfac.codtipdoc inner join Tipo_Pac on docfac.codtipopac=Tipo_Pac.tipopac  inner join concepto on detfac.codcon=concepto.codcon inner join servicio on detfac.codserv=servicio.codserv where docfac.fecha=CTOD("10/04/2014") and docfac.codtipdoc="OPA" AND docfac.codestado="C" and detfac.codcon="39.01.01" and docfac.codtipopac not in ("010","005","008","009")
